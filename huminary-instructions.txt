# Build the chainspec
./target/release/huminary-node build-spec --disable-default-bootnode > huminary-chainspec.json

./target/release/huminary-node export-genesis-wasm --chain raw-huminary-chainspec.json para-1061-wasm

./target/release/huminary-node export-genesis-state --chain raw-huminary-chainspec.json para-1061-genesis-state
